PRD: Kumon Automação Financeira (MVP)

Problema

A gestão financeira do Kumon Camargos (71 alunos, 3 disciplinas, ~60 responsáveis financeiros) é inteiramente manual: emissão de boletos um a um, envio individual por WhatsApp, acompanhamento de pagamentos em planilha, cobrança de inadimplentes por mensagem. Com o crescimento para 80-95 alunos, esse processo não escala e consome ~12h/mês do orientador em tarefas operacionais que deveriam ser automáticas.

Quem Sofre

  • Rafael Pimenta (orientador): perde tempo em operação financeira repetitiva ao invés de focar em pedagógico, retenção e crescimento da unidade
  • Responsáveis financeiros: recebem boletos atrasados, sem consolidação quando têm mais de um filho ou disciplina, sem visibilidade do que estão pagando

Hipóteses

IDHipóteseMétrica
H1Se automatizarmos geração de boletos via Cora API, reduzimos de ~3h para ~5min/mêsTempo de emissão mensal
H2Se consolidarmos boletos por responsável, eliminamos erros de valor e reclamaçõesBoletos com valor incorreto → 0
H3Se enviarmos boleto automaticamente via WhatsApp no momento da geração, aumentamos pontualidade% pagos antes do vencimento
H4Se tivermos dashboard com visão financeira em tempo real, eliminamos planilha manualTempo para visão financeira: 0s

Upsert inteligente: Se responsavel_cpf já existe → atualiza dados do Guardian. Se aluno_nome + guardianId já existe → atualiza. Se disciplina já ativa pro aluno → pula. Assim o CSV pode ser reimportado sem duplicar.

Lógica do Upsert (Server Action)

// actions/students.ts — pseudocódigo

async function importStudents(rows: ParsedRow[]) {
  for (const row of rows) {
    // 1. Upsert Guardian (by CPF)
    const guardian = await prisma.guardian.upsert({
      where: { cpf: cleanCpf(row.responsavel_cpf) },
      update: { name: row.responsavel_nome, whatsapp: row.responsavel_whatsapp },
      create: { name: row.responsavel_nome, cpf: cleanCpf(row.responsavel_cpf),
                whatsapp: row.responsavel_whatsapp, email: row.responsavel_email }
    })

    // 2. Upsert Student (by name + guardianId)
    const student = await prisma.student.upsert({
      where: { name_guardianId: { name: row.aluno_nome, guardianId: guardian.id } },
      update: { ... },
      create: { name: row.aluno_nome, guardianId: guardian.id, status: 'ACTIVE', ... }
    })

    // 3. Upsert Enrollment (by studentId + subject)
    await prisma.enrollment.upsert({
      where: { studentId_subject: { studentId: student.id, subject: mapSubject(row.disciplina) } },
      update: { monthlyFee: row.mensalidade * 100, status: 'ACTIVE' },
      create: { studentId: student.id, subject: mapSubject(row.disciplina),
                monthlyFee: row.mensalidade * 100, status: 'ACTIVE', startDate: new Date() }
    })
  }
}

Autenticação (lib/cora.ts)

class CoraClient {
  private baseUrl = process.env.CORA_BASE_URL
  private clientId = process.env.CORA_CLIENT_ID
  private accessToken: string | null = null
  private tokenExpiresAt: number = 0

  // Certificado mTLS a partir de base64 env var
  private getAgent() {
    const cert = Buffer.from(process.env.CORA_CERT_BASE64!, 'base64')
    const key = Buffer.from(process.env.CORA_KEY_BASE64!, 'base64')
    return new https.Agent({ cert, key })
  }

  async getToken() {
    if (this.accessToken && Date.now() < this.tokenExpiresAt) {
      return this.accessToken
    }
    const res = await fetch(`${this.baseUrl}/token`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: `grant_type=client_credentials&client_id=${this.clientId}`,
      agent: this.getAgent(),
    })
    const data = await res.json()
    this.accessToken = data.access_token
    this.tokenExpiresAt = Date.now() + (data.expires_in * 1000) - 60000
    return this.accessToken
  }

  async createInvoice(params: CreateInvoiceParams) {
    const token = await this.getToken()
    return fetch(`${this.baseUrl}/v2/invoices/`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json',
        'Idempotency-Key': params.idempotencyKey,
      },
      body: JSON.stringify({
        code: params.code,
        customer: {
          name: params.customer.name,
          document: { identity: params.customer.cpf, type: 'CPF' },
          email: params.customer.email,
          address: params.customer.address,
        },
        payment_terms: {
          due_date: params.dueDate, // YYYY-MM-DD
        },
        services: params.items.map(item => ({
          name: item.description,
          amount: item.amount, // centavos
        })),
      }),
      agent: this.getAgent(),
    })
  }
}

Server Action — Gerar Boletos

// actions/billing.ts

'use server'

async function generateMonthlyInvoices(month: string, dueDate: string) {
  const cora = new CoraClient()

  // 1. Busca todos guardians com alunos ativos
  const guardians = await prisma.guardian.findMany({
    where: { students: { some: { status: 'ACTIVE',
      enrollments: { some: { status: 'ACTIVE' } } } } },
    include: { students: { include: { enrollments: true } } }
  })

  const results = []

  for (const guardian of guardians) {
    // 2. Checa se já existe invoice pro mês
    const existing = await prisma.invoice.findFirst({
      where: { guardianId: guardian.id, referenceMonth: month }
    })
    if (existing) { results.push({ guardian, status: 'skipped' }); continue }

    // 3. Calcula items
    const items = guardian.students
      .flatMap(s => s.enrollments
        .filter(e => e.status === 'ACTIVE')
        .map(e => ({
          enrollmentId: e.id,
          description: `${s.name} — ${e.subject}`,
          amount: e.monthlyFee,
        }))
      )

    const totalAmount = items.reduce((sum, i) => sum + i.amount, 0)
    const idempotencyKey = crypto.randomUUID()

    // 4. Chama Cora API
    try {
      const coraRes = await cora.createInvoice({
        idempotencyKey,
        code: `KUMON-${month}-${guardian.cpf}`,
        customer: { name: guardian.name, cpf: guardian.cpf, email: guardian.email },
        dueDate,
        items,
      })
      const coraData = await coraRes.json()

      // 5. Salva no banco
      await prisma.invoice.create({
        data: {
          guardianId: guardian.id,
          coraInvoiceId: coraData.id,
          idempotencyKey,
          referenceMonth: month,
          totalAmount,
          dueDate: new Date(dueDate),
          status: 'OPEN',
          pdfUrl: coraData.pdf_url,
          barcode: coraData.barcode,
          pixQrCode: coraData.pix_qr_code,
          items: { create: items.map(i => ({
            enrollmentId: i.enrollmentId,
            description: i.description,
            amount: i.amount,
          })) }
        }
      })
      results.push({ guardian: guardian.name, status: 'created', amount: totalAmount })
    } catch (err) {
      results.push({ guardian: guardian.name, status: 'error', error: err.message })
    }
  }
  return results
}

// app/api/webhooks/cora/route.ts

export async function POST(req: Request) {
  const body = await req.json()

  // Log raw event
  await prisma.webhookEvent.create({
    data: { coraEventId: body.id, eventType: body.type, payload: body }
  })

  if (body.type === 'invoice.paid') {
    await prisma.invoice.update({
      where: { coraInvoiceId: body.data.invoice_id },
      data: {
        status: 'PAID',
        paidAmount: body.data.paid_amount,
        paidAt: new Date(),
      }
    })
  }

  return Response.json({ received: true })
}

function getWhatsAppLink(guardian: Guardian, invoice: Invoice) {
  const msg = encodeURIComponent(
    `Olá ${guardian.name.split(' ')[0]}! 😊\n\n` +
    `Segue o boleto do Kumon referente a ${invoice.referenceMonth}.\n` +
    `Valor: R$${(invoice.totalAmount / 100).toFixed(2)}\n` +
    `Vencimento: ${format(invoice.dueDate, 'dd/MM/yyyy')}\n\n` +
    `Link do boleto: ${invoice.pdfUrl}\n\n` +
    `Qualquer dúvida, estamos à disposição! 🙏`
  )
  return `https://wa.me/55${guardian.whatsapp}?text=${msg}`
}
